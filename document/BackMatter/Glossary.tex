\newglossaryentry{CAP}{
    name={CAP},
    description={Teorema que establece que en un sistema distribuido no es posible garantizar simultáneamente Consistencia, Disponibilidad y Tolerancia a Particiones. 
    La Consistencia implica que todos los nodos ven los mismos datos al mismo tiempo; 
    la Disponibilidad garantiza que cada solicitud recibe una respuesta, incluso en presencia de fallos; 
    y la Tolerancia a Particiones se refiere a la capacidad del sistema para continuar operando a pesar de fallos de comunicación o desconexiones entre nodos de la red. 
    Por lo tanto, se deben priorizar dos de estas propiedades según el diseño del sistema}

\newglossaryentry{SOLID}
{
  name={SOLID},
  description={Acrónimo de cinco principios de diseño orientado a objetos: 
  \begin{itemize}
  \item \textbf{Responsabilidad Única} (una clase debe tener una sola razón para cambiar), 
  \item \textbf{Abierto/Cerrado} (abierta a extensión, cerrada a modificación), 
  \item \textbf{Sustitución de Liskov} (los objetos de una subclase deben poder sustituir a los de su clase base sin alterar el sistema), 
  \item \textbf{Segregación de Interfaces} (interfaces pequeñas y específicas en lugar de una interfaz grande y genérica), 
  \item \textbf{Inversión de Dependencias} (depender de abstracciones, no de implementaciones concretas). 
  \end{itemize}
  Estos principios fueron recopilados por Robert C. Martin \cite{martin2000} y el acrónimo fue acuñado por Michael Feathers \cite{feathers2004}}
}

\newglossaryentry{hmac-sha256}{
    name={HMAC-SHA256},
    description={Hash-based Message Authentication Code con SHA-256. Mecanismo de criptografía simétrica que verifica simultáneamente la integridad y autenticidad de un mensaje. Combina una clave secreta compartida con el algoritmo de hash SHA-256 para producir un código de autenticación de longitud fija. Cualquier alteración del mensaje o uso de una clave incorrecta genera un código distinto, lo que permite detectar modificaciones no autorizadas. A diferencia de la criptografía asimétrica, emplea una única clave, lo que lo hace computacionalmente eficiente para operaciones como el hashing de contraseñas y la firma de tokens \cite{rfc2104}. SHA-256, estandarizado por el NIST, transforma cualquier entrada en una cadena de 256 bits de la cual resulta computacionalmente inviable reconstruir el dato original \cite{nist_sha256}}
}
